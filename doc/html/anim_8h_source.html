132 template <class T> void set_exec_cb(ExecCb<T> exec_cb) {
133 // the exec_cb callback takes a void* as 1st parameter, in order to
134 // accommodate pointers of any type; this means the animation object
135 // isn't passed to the callback; to make the callback accept a reference
136 // to a C++ class instance, we need to be able to pass a reference
137 // to the callback function somehow. For callbacks with lv_anim_t* first
138 // params, it's easy: set user_data to this and retrieve this in callback.
139 // For this one, this is trickier. My approach is to provide a callback
140 // container that also stores the variable that would normally be passed
141 // to exec_cb. Then we change the variable in the lv_anim_t object to
142 // a pointer to the callback container. We can then access the callback
143 // and the variable upon which it acts.
144 lv_anim_set_var(this->raw_ptr(), static_cast<void*>(this));
145 using ExecCbType = Callback<void, T&, int32_t>;
146 this->exec_cb = std::make_shared<ExecCbType>(exec_cb);
147 // definition of callback wrapper function; this will call C++ callback
148 // with stored var
149 auto f = [](void * lv_obj, int32_t value) {
150 auto anim = reinterpret_cast<Animation*>(lv_obj);
151 auto cb = *std::dynamic_pointer_cast<ExecCbType>(anim->exec_cb);
152 cb(anim->get_var<T>(), value);
153 };
154 lv_anim_set_exec_cb(this->raw_ptr(), f);
155 }
